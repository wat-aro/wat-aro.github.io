(define (fixed-point2 f first-guess)
  (define (close-enough? v1 v2)
    (< (abs (- v1 v2)) tolerance))
  (define (try guess)
    (let ((next (f guess)))
      (display guess)
      (newline)
      (if (close-enough? guess next)
          next
          (try next))))
  (try first-guess))

(define (my-exp)
  (fixed-point2 (lambda (x) (/ (log 1000) (log x)))
                2.0))

(define (my-exp2)
  (fixed-point2 (lambda (x) (average x (/ (log 1000) (log x))))
                2.0))
