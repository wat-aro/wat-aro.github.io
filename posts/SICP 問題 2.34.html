(define (accumulate op initial sequence)
  (if (null? sequence)
      initial
      (op (car sequence)
          (accumulate op initial (cdr sequence)))))

(define (horner-eval x coefficient-sequence)
  (define (iter lis res)
    (cond ((null? lis) res)
          (else (iter (cdr lis) (+ (car lis) (* x res))))))
  (iter (reverse coefficient-sequence) 0))
