(define (element-of-set? x set)
  (cond ((null? set) false)
        ((equal? x (car set)) true)
        (else (element-of-set? x (cdr set)))))

(define (unionset s t)
  (cond ((null? s) t)
        ((element-of-set? (car s) t)
         (unionset (cdr s) t))
        (else (cons (car s)
                    (unionset (cdr s) t)))))
